### API de Likes para Free Fire com Website Simplificado

Vamos simplificar o website: Adicionei CSS para estilização (usando Bootstrap para rapidez e beleza), removi login/registro, e mantive apenas um botão "Gerar API Key" que cria uma API key anônima (usuário temporário no banco). O site é público, qualquer um pode gerar keys. Tokens ainda são salvos no banco.

**Mudanças**:
- Website com CSS (Bootstrap) para layout moderno.
- Botão único para gerar API key via endpoint /generate-api-key.
- API keys são geradas automaticamente sem autenticação.
- Atualizei server.js, removendo /register e /login.

#### 1. Atualizações no package.json
Adicione Bootstrap via CDN (não precisa instalar, usamos link).

#### 2. server.js Atualizado
```javascript
const express = require('express');
const axios = require('axios');
const cors = require('cors');
const mongoose = require('mongoose');
const path = require('path');
const crypto = require('crypto');
require('dotenv').config();

const User = require('./models/User');
const Token = require('./models/Token');

const app = express();
const PORT = process.env.PORT || 3000;

// Conectar ao MongoDB
mongoose.connect(process.env.MONGO_URI, {
  useNewUrlParser: true,
  useUnifiedTopology: true,
}).then(() => console.log('MongoDB conectado')).catch(err => console.error(err));

app.use(cors());
app.use(express.json());
app.use(express.static(path.join(__dirname, 'public')));

// Middleware para verificar API key
const authenticateApiKey = async (req, res, next) => {
  const apiKey = req.header('x-api-key');
  if (!apiKey) return res.status(401).json({ success: false, message: 'API key obrigatória.' });
  const user = await User.findOne({ apiKey });
  if (!user) return res.status(401).json({ success: false, message: 'API key inválida.' });
  req.user = user;
  next();
};

// Novo endpoint: Gerar API key anônima
app.post('/generate-api-key', async (req, res) => {
  try {
    const apiKey = crypto.randomUUID();
    const user = new User({
      username: `anon-${Date.now()}`,  // Username temporário
      email: `anon-${Date.now()}@temp.com`,  // Email temporário
      password: 'temp',  // Senha dummy (não usada)
      apiKey,
    });
    await user.save();
    res.json({ success: true, apiKey });
  } catch (err) {
    res.status(500).json({ success: false, message: 'Erro ao gerar API key.' });
  }
});

// Endpoint para enviar likes
app.post('/send-likes', authenticateApiKey, async (req, res) => {
  const { uid, region, accessToken } = req.body;

  if (!uid || !region || !accessToken) {
    return res.status(400).json({ success: false, message: 'UID, região e accessToken são obrigatórios.' });
  }
  if (!/^\d+$/.test(uid)) {
    return res.status(400).json({ success: false, message: 'UID deve ser numérico.' });
  }
  const validRegions = ['BR', 'NA', 'SA', 'EU', 'AS', 'OC'];
  if (!validRegions.includes(region.toUpperCase())) {
    return res.status(400).json({ success: false, message: 'Região inválida.' });
  }

  // Salva ou atualiza o token
  await Token.findOneAndUpdate(
    { userId: req.user._id },
    { accessToken, expiresAt: new Date(Date.now() + 24 * 60 * 60 * 1000) },
    { upsert: true, new: true }
  );

  const apiUrl = `https://getvyenx.cloud/free-fire/send-likes?uid=${uid}&apikey=${req.user.apiKey}&region=${region.toUpperCase()}&access_token=${accessToken}`;

  try {
    const response = await axios.get(apiUrl, { timeout: 10000 });
    const data = response.data;
    if (data.success) {
      res.json({ success: true, message: `Likes enviados para UID ${uid}.` });
    } else {
      res.status(400).json({ success: false, message: data.message || 'Erro na API.' });
    }
  } catch (error) {
    res.status(500).json({ success: false, message: `Erro: ${error.message}` });
  }
});

// Endpoint para obter token salvo
app.get('/get-token', authenticateApiKey, async (req, res) => {
  const token = await Token.findOne({ userId: req.user._id });
  if (!token) return res.status(404).json({ success: false, message: 'Nenhum token salvo.' });
  res.json({ success: true, accessToken: token.accessToken });
});

app.get('/health', (req, res) => {
  res.json({ status: 'OK' });
});

app.listen(PORT, () => {
  console.log(`API rodando na porta ${PORT}`);
});
```

#### 3. public/index.html Atualizado (Com CSS e Botão)
```html
<!DOCTYPE html>
<html lang="pt-BR">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>API de Likes Free Fire</title>
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">
  <style>
    body { background-color: #f8f9fa; }
    .container { max-width: 600px; margin-top: 50px; }
    .card { box-shadow: 0 4px 8px rgba(0,0,0,0.1); }
    #apiKeyDisplay { font-weight: bold; color: #28a745; }
  </style>
</head>
<body>
  <div class="container">
    <div class="card">
      <div class="card-header bg-primary text-white">
        <h1 class="h4 mb-0">Gerador de API Key para Likes Free Fire</h1>
      </div>
      <div class="card-body">
        <p>Clique no botão para gerar uma API key gratuita e usar a API de likes.</p>
        <button id="generateBtn" class="btn btn-success btn-lg">Gerar API Key</button>
        <div id="result" class="mt-3" style="display: none;">
          <p>Sua API Key: <span id="apiKeyDisplay"></span></p>
          <p>Use no header <code>x-api-key</code> para chamadas à API.</p>
        </div>
      </div>
    </div>
  </div>
  <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/js/bootstrap.bundle.min.js"></script>
  <script>
    document.getElementById('generateBtn').addEventListener('click', async () => {
      const btn = document.getElementById('generateBtn');
      btn.disabled = true;
      btn.textContent = 'Gerando...';
      try {
        const response = await fetch('/generate-api-key', { method: 'POST' });
        const data = await response.json();
        if (data.success) {
          document.getElementById('apiKeyDisplay').textContent = data.apiKey;
          document.getElementById('result').style.display = 'block';
        } else {
          alert('Erro: ' + data.message);
        }
      } catch (error) {
        alert('Erro ao gerar API key.');
      } finally {
        btn.disabled = false;
        btn.textContent = 'Gerar API Key';
      }
    });
  </script>
</body>
</html>
```

#### Como Usar
- Execute `npm start`.
- Acesse `http://localhost:3000`: Site com botão para gerar API key.
- Clique para gerar, copie a key.
- Use no Postman: Header `x-api-key: sua-key`, POST /send-likes com body JSON.
